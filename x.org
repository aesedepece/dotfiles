* xinitrc
:PROPERTIES:
:tangle: ~/.xinitrc
:shebang: #!/bin/sh
:END:

** Custom fonts
#+BEGIN_SRC sh
  xset +fp /usr/share/fonts
  xset +fp /usr/share/fonts/TTF/
  xset fp rehash
#+END_SRC

** Disable bell
#+BEGIN_SRC sh
  xset -b
#+END_SRC

** Set the cursor
#+BEGIN_SRC sh
  xsetroot -cursor_name left_ptr
#+END_SRC

** Map ctrl to capslock
#+BEGIN_SRC sh
  setxkbmap -option ctrl:nocaps
#+END_SRC

** Launch daemons
#+BEGIN_SRC sh
  if [ -d /etc/X11/xinit/xinitrc.d ]; then
    for f in /etc/X11/xinit/xinitrc.d/*; do
      [ -x "$f" ] && . "$f"
    done
    unset f
  fi
#+END_SRC

** Load .Xresources
#+BEGIN_SRC sh
  [[ -f ~/.Xresources ]] && xrdb -merge ~/.Xresources
#+END_SRC

** Launch window manager
#+BEGIN_SRC sh
  stumpwm
#+END_SRC

* xresources
:PROPERTIES:
:tangle: ~/.Xresources
:END:
